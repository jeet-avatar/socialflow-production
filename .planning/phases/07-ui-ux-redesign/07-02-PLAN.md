    setScriptModel(config.script_model ?? '');

    setVideoBgProvider(config.video_bg_provider ?? '');
  }

**handleSave:**
```typescript
const handleSave = async () => {
  setSaving(true);
  try {
    await updateConfig({
      channel_id: channelId,
      script_model: scriptModel,
      voice_provider: voiceProvider,
      video_bg_provider: videoBgProvider,
    });
    setSaveSuccess(true);
    setTimeout(() => setSaveSuccess(false), 2000);
  } finally {
    setSaving(false);
  }
};
```

Note: `updateConfig` from useModelConfig calls POST /model-config (upsert — no PUT route exists, Research pitfall 5).
