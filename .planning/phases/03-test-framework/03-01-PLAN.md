```python
from unittest.mock import patch, MagicMock

def test_video_delete_calls_s3(client, mock_db, auth_headers):
    """Deleting a video should call delete_s3_object."""
    # Seed a video in mock DB
    from bson import ObjectId
    video_id = str(ObjectId())
    mock_db["videos"].insert_one({
        "_id": ObjectId(video_id),
        "user_id": "dev_user",
        "title": "Test Video",
        "s3_bucket": "my-bucket",
        "s3_key": "videos/test.mp4",
    })

    with patch("utils.s3_service.delete_s3_object") as mock_s3:
        mock_s3.return_value = True
        response = client.delete(f"/videos/{video_id}", headers=auth_headers)
        assert response.status_code == 200
        mock_s3.assert_called_once_with(bucket="my-bucket", key="videos/test.mp4")

def test_video_delete_continues_if_s3_fails(client, mock_db, auth_headers):
    """S3 failure should not block MongoDB delete."""
    from bson import ObjectId
    video_id = str(ObjectId())
    mock_db["videos"].insert_one({
        "_id": ObjectId(video_id),
        "user_id": "dev_user",
        "title": "Test Video",
        "s3_bucket": "my-bucket",
        "s3_key": "videos/test.mp4",
    })

    with patch("utils.s3_service.delete_s3_object") as mock_s3:
        mock_s3.return_value = False  # S3 failed
        response = client.delete(f"/videos/{video_id}", headers=auth_headers)
        assert response.status_code == 200
        assert response.json()["s3_deleted"] is False
        # Video should still be deleted from DB
        assert mock_db["videos"].find_one({"_id": ObjectId(video_id)}) is None
```
